  private failures = 0;
  private readonly threshold = 5;
  private readonly cooldown = 3600000; // 1 hour
  
  async execute(fn: Function) {
    if (this.failures >= this.threshold) {
      if (!this.shouldReset()) {
        throw new Error('Circuit breaker is open');
      }
      this.reset();
    }
    
    try {
      const result = await fn();
      this.onSuccess();
      return result;
    } catch (error) {
      this.onFailure();
      throw error;
    }
  }
}
